resource "azurerm_storage_account" "storage_accounts" {
  for_each = {
    for sa in var.storage_accounts : sa.name => sa
    if sa.is_active
  }

  name                     = each.value.name
  resource_group_name      = var.resource_group_name
  location                 = var.location
  account_tier             = each.value.account_tier
  account_replication_type = each.value.account_replication_type
  account_kind             = each.value.account_kind
  access_tier              = each.value.access_tier

  min_tls_version                 = "TLS1_2"
  allow_nested_items_to_be_public = false

  tags = var.common_tags
}

locals {
  active_container_map = {
    for item in flatten([
      for sa in var.storage_accounts : [
        for c in sa.containers : {
          key                   = "${sa.name}-${c.name}"
          storage_account_name  = sa.name
          container_name        = c.name
          container_access_type = c.container_access_type
          metadata              = c.metadata
          sa_is_active          = sa.is_active
          container_is_active   = c.is_active
        }
      ]
    ]) : item.key => item
    if item.sa_is_active && item.container_is_active
  }
}

resource "azurerm_storage_container" "containers" {
  for_each = local.active_container_map

  name                  = each.value.container_name
  storage_account_name  = each.value.storage_account_name
  container_access_type = each.value.container_access_type
  metadata              = each.value.metadata
}
