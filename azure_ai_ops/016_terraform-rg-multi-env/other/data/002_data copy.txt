variable "use_existing_workspace" {
  type    = bool
  default = true
}

variable "workspace_name" {
  type = string
}

data "azurerm_log_analytics_workspace" "existing" {
  count               = var.use_existing_workspace ? 1 : 0
  name                = var.workspace_name
  resource_group_name = var.resource_group_name
}

resource "azurerm_log_analytics_workspace" "new" {
  count               = var.use_existing_workspace ? 0 : 1
  name                = var.workspace_name
  location            = var.location
  resource_group_name = var.resource_group_name
  sku                 = "PerGB2018"
  retention_in_days   = 30
}

locals {
  workspace_id = var.use_existing_workspace
    ? data.azurerm_log_analytics_workspace.existing[0].id
    : azurerm_log_analytics_workspace.new[0].id
}
